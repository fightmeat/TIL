```python
import random

# Oracle 연결 문자열: username/password@host:port/service_name
DB_CONNECTION_STRING = "your_username/your_password@your_host:your_port/your_service_name"

def setup_database():
    conn = cx_Oracle.connect(DB_CONNECTION_STRING)
    cursor = conn.cursor()
    cursor.execute("""
    CREATE TABLE IF NOT EXISTS vocabulary (
        word VARCHAR2(255) PRIMARY KEY,
        meanings VARCHAR2(1000)
    )
    """)
    conn.commit()
    conn.close()

def load_data():
    conn = cx_Oracle.connect(DB_CONNECTION_STRING)
    cursor = conn.cursor()
    cursor.execute("SELECT word, meanings FROM vocabulary")
    data = {row[0]: row[1].split(', ') for row in cursor.fetchall()}
    conn.close()
    return data

def save_data(vocabulary):
    conn = cx_Oracle.connect(DB_CONNECTION_STRING)
    cursor = conn.cursor()

    for word, meanings in vocabulary.items():
        meanings_str = ', '.join(meanings)
        cursor.execute("MERGE INTO vocabulary USING DUAL ON (word = :word) WHEN MATCHED THEN UPDATE SET meanings = :meanings WHEN NOT MATCHED THEN INSERT (word, meanings) VALUES (:word, :meanings)", {'word': word, 'meanings': meanings_str})

    conn.commit()
    conn.close()
```
